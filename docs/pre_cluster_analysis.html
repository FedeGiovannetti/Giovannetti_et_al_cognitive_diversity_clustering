## Function for efficient calculation

alpha_calc = function(dimension, data){
  
  capture.output({
    alpha_out <- data %>% 
      select(all_of(dimension)) %>% 
      psych::alpha() %>% 
      suppressMessages() %>% 
      suppressWarnings()
  
})
  
  alpha_df = alpha_out$total %>% 
    # mutate(Dimension = names(dimension)) %>% 
    select(raw_alpha, std.alpha) %>% 
    rename(Alpha = raw_alpha,
           'Alpha (std)' = std.alpha)
  
 return(alpha_df)
  
}

## Table output


map(.x = cbq_dimensions, .f = ~alpha_calc(.x, cbq_items)) %>%
  list_rbind(names_to = "Dimension") %>% 
    remove_rownames() %>% 
  mutate(Dimension = stringr::str_replace(Dimension, "_", " "))%>%
    # flextable() %>% 
    custom_flextable("Cronbach's Alpha for CBQ") %>% 
 
    autofit()
